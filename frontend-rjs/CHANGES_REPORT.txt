  export async function getIdToken() {
    const user = auth.currentUser;
    if (!user) throw new Error("No user is currently signed in.");

  }

  // ERROR PARSING — converts Firebase error codes to user-friendly messages
  export function getAuthErrorMessage(errorCode) {
    const errorMap = {
      "auth/email-already-in-use": "This email is already registered.",
      "auth/invalid-email": "Please enter a valid email address.",
      "auth/weak-password": "Password must be at least 6 characters.",
      "auth/wrong-password": "Incorrect password.",
      "auth/invalid-credential": "Invalid email or password.",
      // ... etc
    };
    return errorMap[errorCode] || "An unexpected error occurred.";
  }

KEY CODE:
  export function AuthProvider({ children }) {
    const [user, setUser] = useState(null);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
      // Firebase listener — fires when auth state changes
      const unsubscribe = onAuthChange(async (firebaseUser) => {
        if (firebaseUser) {
          const token = await firebaseUser.getIdToken();
          console.log("🔑 Firebase ID Token:", token);
          setUser(firebaseUser);
        } else {
          setUser(null);
        }
        setLoading(false);
      });
      return () => unsubscribe();
    }, []);

    // Exposed via useAuth() hook
    const value = { user, loading, logout, refreshToken, isAuthenticated: !!user };
    return <AuthContext.Provider value={value}>{children}</AuthContext.Provider>;
  }

  export function useAuth() {
    return useContext(AuthContext);
  }

  export function useToast() {
    const [toast, setToast] = useState({ message: "", type: "success" });
    const showToast = (message, type = "success", duration = 3000) => {
      setToast({ message, type });
      setTimeout(() => setToast({ message: "", type: "success" }), duration);
    };
    return { toast, showToast, hideToast };
  }
